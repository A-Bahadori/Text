void Main()
{
    // دریافت مسیر از کاربر
    var path = Input.GetDirectory("مسیر مورد نظر را انتخاب کنید:");
    if (string.IsNullOrEmpty(path) || !Directory.Exists(path))
    {
        "مسیر معتبر نیست!".Dump();
        return;
    }
    
    // جستجوی تمام پوشه‌های obj و bin به صورت بازگشتی
    var dirsToDelete = Directory.GetDirectories(path, "*", SearchOption.AllDirectories)
                                .Where(d => 
                                    Path.GetFileName(d).Equals("obj", StringComparison.OrdinalIgnoreCase) ||
                                    Path.GetFileName(d).Equals("bin", StringComparison.OrdinalIgnoreCase))
                                .ToList();
    
    if (!dirsToDelete.Any())
    {
        "هیچ پوشه obj یا bin یافت نشد.".Dump();
        return;
    }
    
    // حذف پوشه‌ها
    int deletedCount = 0;
    foreach (var dir in dirsToDelete)
    {
        try
        {
            Directory.Delete(dir, true);
            deletedCount++;
        }
        catch (Exception ex)
        {
            $"خطا در حذف {dir}: {ex.Message}".Dump();
        }
    }
    
    $"{deletedCount} پوشه با موفقیت حذف شد.".Dump();
}
