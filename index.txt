[SetUp]
public void Setup()
{
    _tempFolder = Path.Combine(Path.GetTempPath(), Guid.NewGuid().ToString());
    Directory.CreateDirectory(_tempFolder);
    _testFileName = "test.docx";
    _testFilePath = Path.Combine(_tempFolder, _testFileName);
    
    // ساختن فایل Word با متن فارسی و فونت نستعلیق
    var doc = new Document();
    var builder = new DocumentBuilder(doc);
    
    // ایجاد یک Shape برای متن فارسی
    Shape shape = new Shape(doc, ShapeType.TextBox)
    {
        Width = 200,
        Height = 100,
        HorizontalAlignment = HorizontalAlignment.Center
    };
    
    shape.AppendChild(new Paragraph(doc));
    
    // تنظیم فونت نستعلیق برای Run
    Run run = new Run(doc, "متن فارسی با فونت نستعلیق");
    run.Font.Name = "IranNastaliq"; // نام فونت نستعلیق
    run.Font.Size = 18;
    
    shape.FirstParagraph.AppendChild(run);
    
    // تنظیم جهت متن برای پشتیبانی از فارسی
    shape.FirstParagraph.ParagraphFormat.Bidi = true;
    
    builder.InsertNode(shape);
    doc.Save(_testFilePath);
}
