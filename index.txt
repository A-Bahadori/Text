[SetUp]
public void Setup()
{
    _tempFolder = Path.Combine(Path.GetTempPath(), Guid.NewGuid().ToString());
    Directory.CreateDirectory(_tempFolder);
    _testFileName = "test.docx";
    _testFilePath = Path.Combine(_tempFolder, _testFileName);
    
    // ساختن فایل Word با متن فارسی و فونت نستعلیق
    var doc = new Document();
    var builder = new DocumentBuilder(doc);
    
    // تنظیم فونت نستعلیق
    builder.Font.Name = "IranNastaliq"; // نام فونت نستعلیق
    builder.Font.Size = 18;
    
    // تنظیم جهت متن برای پشتیبانی از فارسی
    builder.ParagraphFormat.Bidi = true;
    
    // درج متن فارسی
    builder.Write("متن فارسی با فونت نستعلیق");
    
    // ذخیره فایل
    doc.Save(_testFilePath);
}
