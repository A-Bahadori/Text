<system.webServer>
    <!-- فعال کردن اجرای تمام ماژول‌های مدیریت شده برای تمام درخواست‌ها -->
    <modules runAllManagedModulesForAllRequests="true">
      <!-- حذف ماژول WebDAV در صورت نصب -->
      <remove name="WebDAVModule" />
    </modules>

    <!-- پیکربندی Handlerها جهت پشتیبانی از URLهای بدون پسوند -->
    <handlers>
      <!-- حذف Handlerهای مربوط به WebDAV -->
      <remove name="WebDAV" />

      <!-- حذف Handler قدیمی ExtensionlessUrlHandler و اضافه کردن تنظیمات به‌روز -->
      <remove name="ExtensionlessUrlHandler-Integrated-4.0" />
      <add name="ExtensionlessUrlHandler-Integrated-4.0" 
           path="*." 
           verb="GET,HEAD,POST,DEBUG,PUT,DELETE,PATCH,OPTIONS" 
           type="System.Web.Handlers.TransferRequestHandler" 
           preCondition="integratedMode,runtimeVersionv4.0" />
    </handlers>

    <!-- تنظیمات امنیتی و فیلتر کردن درخواست‌ها -->
    <security>
      <requestFiltering>
        <hiddenSegments>
          <!-- حذف پوشه bin از لیست پوشه‌های پنهان (در صورت نیاز) -->
          <remove segment="bin" />
        </hiddenSegments>
      </requestFiltering>
    </security>
  </system.webServer>
