using System;
using System.IO;

namespace Ray.BPMS.Infrastructure.Helpers
{
    /// <summary>
    /// فعال‌سازی لایسنس Aspose.Words
    /// </summary>
    public static class AsposeWordsLicenseActivator
    {
        /// <summary>
        /// فعال‌سازی لایسنس Aspose.Words با استفاده از فایل
        /// </summary>
        public static void ActivateLicense()
        {
            try
            {
                // روش اول: استفاده از LicenseHelper موجود
                // فعال‌سازی همه لایسنس‌های Aspose از جمله Aspose.Words
                LicenseHelper.ActivateMemoryPatching();
            }
            catch (Exception ex)
            {
                // اگر روش اول با مشکل مواجه شد، از روش استاندارد استفاده می‌کنیم
                try
                {
                    // مسیر فایل لایسنس
                    string licenseFilePath = Path.Combine(AppDomain.CurrentDomain.BaseDirectory, "Lib", "Aspose.Total.NET.lic");
                    
                    // اگر فایل در مسیر اصلی پیدا نشد، از مسیر نسبی استفاده می‌کنیم
                    if (!File.Exists(licenseFilePath))
                    {
                        licenseFilePath = @"..\..\Lib\Aspose.Total.NET.lic";
                    }

                    // فعال‌سازی فقط لایسنس Aspose.Words
                    var wordLicense = new Aspose.Words.License();
                    wordLicense.SetLicense(licenseFilePath);
                }
                catch (Exception innerEx)
                {
                    // لاگ خطا
                    System.Diagnostics.Debug.WriteLine($"خطا در فعال‌سازی لایسنس Aspose.Words: {ex.Message}");
                    System.Diagnostics.Debug.WriteLine($"جزئیات خطا: {innerEx.Message}");
                    
                    // می‌توانید خطا را به سیستم لاگ خود ارسال کنید یا آن را مجدداً پرتاب کنید
                    // throw new Exception("خطا در فعال‌سازی لایسنس Aspose.Words", innerEx);
                }
            }
        }
    }
}
