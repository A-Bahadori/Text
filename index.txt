    // کلاس مدیریت لاگر با استفاده از الگوی Singleton
    public static class SILogger
    {
        private static readonly SiAuto _siAuto;
        
        static SILogger()
        {
            _siAuto = new SiAuto();
            
            // تنظیمات پیش‌فرض
            _siAuto.Si.Connections = "file(filename=\"C:\\Logs\\AppLog.sil\", rotate=daily)";
            _siAuto.Si.Level = Level.Debug;
            _siAuto.Si.Enabled = true;
        }
        
        public static Session MainSession
        {
            get { return _siAuto.Main; }
        }
        
        public static void Configure(string connections, Level level = Level.Debug)
        {
            _siAuto.Si.Connections = connections;
            _siAuto.Si.Level = level;
        }
        
        public static void SetEnabled(bool enabled)
        {
            _siAuto.Si.Enabled = enabled;
        }
        
        public static void LogMethodEntry(string methodName)
        {
            _siAuto.Main.EnterMethod(methodName);
        }
        
        public static void LogMethodExit(string methodName)
        {
            _siAuto.Main.LeaveMethod(methodName);
        }
        
        public static void LogInfo(string message)
        {
            _siAuto.Main.LogMessage(message);
        }
        
        public static void LogWarning(string message)
        {
            _siAuto.Main.LogWarning(message);
        }
        
        public static void LogError(string message, Exception ex = null)
        {
            if (ex != null)
            {
                _siAuto.Main.LogException(message, ex);
            }
            else
            {
                _siAuto.Main.LogError(message);
            }
        }
        
        public static void LogDebug(string message)
        {
            _siAuto.Main.LogDebug(message);
        }
        
        public static void LogObject(string title, object value)
        {
            // چون در نسخه قدیمی LogObject به صورت مستقیم وجود ندارد، می‌توانیم خصوصیات را به صورت دستی لاگ کنیم
            _siAuto.Main.LogMessage($"--- {title} ---");
            
            if (value == null)
            {
                _siAuto.Main.LogMessage("(null)");
                return;
            }
            
            foreach (var prop in value.GetType().GetProperties())
            {
                try
                {
                    var propValue = prop.GetValue(value, null);
                    _siAuto.Main.LogString($"{prop.Name}", propValue?.ToString() ?? "(null)");
                }
                catch
                {
                    _siAuto.Main.LogString($"{prop.Name}", "(error getting value)");
                }
            }
        }
        
        public static void BeginTimedOperation(string operationName)
        {
            _siAuto.Main.EnterMethod(operationName);
            _siAuto.Main.LogValue("Start Time", DateTime.Now);
        }
        
        public static void EndTimedOperation(string operationName, DateTime startTime)
        {
            TimeSpan elapsed = DateTime.Now - startTime;
            _siAuto.Main.LogValue("Elapsed Time (ms)", elapsed.TotalMilliseconds);
            _siAuto.Main.LeaveMethod(operationName);
        }
    }

    public class Customer
    {
        public int Id { get; set; }
        public string Name { get; set; }

    }

    }
    
    public class Order
    {

    // کلاس‌های سرویس دهنده برای مثال‌های کاربردی
    public class DataService
    {
        public Customer GetCustomer(int id)
        {
            SILogger.LogMethodEntry("GetCustomer");
            try 
            {
                SILogger.LogInteger("Customer ID", id);
                
                // شبیه‌سازی دریافت مشتری از دیتابیس
                if (id <= 0)
                {
                    SILogger.LogWarning("Invalid customer ID provided");
                    return null;
                }
                
                Thread.Sleep(100); // شبیه‌سازی تأخیر دسترسی به دیتابیس
                
                var customer = new Customer
                {
                    Id = id,
                    Name = $"مشتری {id}",
                    Email = $"customer{id}@example.com",
                    Phone = $"09{id.ToString().PadLeft(9, '1')}"
                };
                
                SILogger.LogObject("Retrieved Customer", customer);
                return customer;
            }
            catch (Exception ex)
            {
                SILogger.LogError("Error retrieving customer", ex);
                throw;
            }
            finally
            {
                SILogger.LogMethodExit("GetCustomer");
            }
        }
        
        public List<Product> GetProducts(int count)
        {
            SILogger.LogMethodEntry("GetProducts");
            try
            {
                SILogger.LogInteger("Requested Count", count);
                
                if (count <= 0)
                {
                    SILogger.LogWarning("Invalid product count requested");
                    return new List<Product>();
                }
                
                var products = new List<Product>();
                for (int i = 1; i <= count; i++)
                {
                    Thread.Sleep(50); // شبیه‌سازی تأخیر
                    
                    var product = new Product
                    {
                        Id = i,
                        Name = $"محصول {i}",
                        Price = i * 10000,
                        Quantity = 10
                    };
                    
                    products.Add(product);
                    SILogger.LogString("Added Product", product.Name);
                }
                
                SILogger.LogInteger("Products Retrieved", products.Count);
                return products;
            }
            catch (Exception ex)
            {
                SILogger.LogError("Error retrieving products", ex);
                throw;
            }
            finally
            {
                SILogger.LogMethodExit("GetProducts");
            }
        }
        
        public bool SaveOrder(Order order)
        {
            SILogger.LogMethodEntry("SaveOrder");
            DateTime startTime = DateTime.Now;
            
            try
            {
                if (order == null)
                {
                    SILogger.LogError("Null order provided");
                    return false;
                }
                
                SILogger.LogObject("Order Info", order);
                SILogger.LogObject("Customer Info", order.Customer);
                
                SILogger.LogMessage("Saving products:");
                foreach (var product in order.Products)
                {
                    SILogger.LogObject($"Product {product.Id}", product);
                    Thread.Sleep(20); // شبیه‌سازی ذخیره محصول
                }
                
                // شبیه‌سازی ذخیره سفارش
                Thread.Sleep(200);
                
                SILogger.LogMessage("Order saved successfully");
                return true;
            }
            catch (Exception ex)
            {
                SILogger.LogError("Error saving order", ex);
                return false;
            }
            finally
            {
                TimeSpan elapsed = DateTime.Now - startTime;
                SILogger.LogValue("Save Operation Duration (ms)", elapsed.TotalMilliseconds);
                SILogger.LogMethodExit("SaveOrder");
            }
        }
    }
    
    public class FileProcessor
    {
        public void ProcessFile(string filePath)
        {
            SILogger.LogMethodEntry("ProcessFile");
            
            try
            {
                SILogger.LogString("File Path", filePath);
                
                if (string.IsNullOrEmpty(filePath))
                {
                    SILogger.LogError("Empty file path provided");
                    return;
                }
                
                if (!File.Exists(filePath))
                {
                    SILogger.LogError($"File not found: {filePath}");
                    return;
                }
                
                // لاگ اطلاعات فایل
                var fileInfo = new FileInfo(filePath);
                SILogger.LogValue("File Size (bytes)", fileInfo.Length);
                SILogger.LogValue("Creation Time", fileInfo.CreationTime);
                
                // لاگ محتوای فایل (برای فایل‌های متنی)
                if (fileInfo.Extension.ToLower() == ".txt" || fileInfo.Extension.ToLower() == ".log")
                {
                    SILogger.MainSession.LogFile("File Content", filePath);
                }
                
                // شبیه‌سازی پردازش فایل
                SILogger.LogMessage("Processing file contents...");
                Thread.Sleep(300);
                
                SILogger.LogMessage("File processing completed");
            }
            catch (Exception ex)
            {
                SILogger.LogError("Error processing file", ex);
            }
            finally
            {
                SILogger.LogMethodExit("ProcessFile");
            }
        }
    }

        private static void FileOperationsExample()
        {
            _si.Main.EnterMethod("FileOperationsExample");
            
            try
            {
                _si.Main.LogMessage("مثال ۵: عملیات فایل");
                
                // ایجاد یک فایل متنی موقت برای تست
                string tempFilePath = Path.Combine(Path.GetTempPath(), "smartinspect_test.txt");
                
                try
                {
                    // نوشتن در فایل
                    _si.Main.LogString("مسیر فایل", tempFilePath);
                    _si.Main.LogMessage("در حال ایجاد فایل...");
                    
                    File.WriteAllText(tempFilePath, "این یک فایل تست برای SmartInspect است.\r\n" +
                                                  "خط دوم فایل تست.\r\n" +
                                                  "خط سوم فایل تست.");
                    
                    _si.Main.LogMessage("فایل با موفقیت ایجاد شد");
                    
                    // نمایش اطلاعات فایل
                    FileInfo fileInfo = new FileInfo(tempFilePath);
                    _si.Main.LogValue("اندازه فایل (بایت)", fileInfo.Length);
                    _si.Main.LogValue("زمان ایجاد", fileInfo.CreationTime);
                    
                    // لاگ کردن محتوای فایل
                    _si.Main.LogFile("محتوای فایل", tempFilePath);
                    
                    // خواندن و ثبت محتوای فایل به صورت دستی
                    string[] lines = File.ReadAllLines(tempFilePath);
                    _si.Main.LogInteger("تعداد خطوط", lines.Length);
                    
                    for (int i = 0; i < lines.Length; i++)
                    {
                        _si.Main.LogString($"خط {i+1}", lines[i]);
                    }
                    
                    // پردازش فایل با استفاده از کلاس FileProcessor
                    _fileProcessor.ProcessFile(tempFilePath);
                }
                finally
                {
                    // پاک کردن فایل موقت
                    if (File.Exists(tempFilePath))
                    {
                        File.Delete(tempFilePath);
                        _si.Main.LogMessage("فایل موقت حذف شد");
                    }
                }
            }
            catch (Exception ex)
            {
                _si.Main.LogException("خطا در عملیات فایل", ex);
            }
            finally
            {
                _si.Main.LeaveMethod("FileOperationsExample");
            }
        }
        
        private static void DatabaseOperationSimulationExample()
        {
            _si.Main.EnterMethod("DatabaseOperationSimulationExample");
            
            try
            {
                _si.Main.LogMessage("مثال ۶: شبیه‌سازی عملیات دیتابیس");
                
                // شبیه‌سازی یک عملیات دیتابیس کامل
                
                // 1. دریافت مشتری
                Customer customer = _dataService.GetCustomer(123);
                
                if (customer == null)
                {
                    _si.Main.LogError("مشتری یافت نشد");
                    return;
                }
                
                // 2. دریافت محصولات
                List<Product> products = _dataService.GetProducts(3);
                
                if (products.Count == 0)
                {
                    _si.Main.LogError("محصولی یافت نشد");
                    return;
                }
                
                // 3. ایجاد سفارش
                var order = new Order
                {
                    Id = 1001,
                    Customer = customer,
                    Products = products,
                    OrderDate = DateTime.Now
                };
                
                // 4. محاسبه و لاگ کردن مجموع سفارش
                decimal total = 0;
                _si.Main.LogMessage("محاسبه مجموع سفارش:");
                
                foreach (var product in order.Products)
                {
                    decimal itemTotal = product.Price * product.Quantity;
                    _si.Main.LogValue($"محصول {product.Id}: {product.Name}", itemTotal);
                    total += itemTotal;
                }
                
                _si.Main.LogValue("مجموع نهایی", total);
                
                // 5. ذخیره سفارش
                bool saveResult = _dataService.SaveOrder(order);
                
                _si.Main.LogBoolean("نتیجه ذخیره سفارش", saveResult);
                
                if (saveResult)
                {
                    _si.Main.LogMessage("سفارش با موفقیت ثبت شد");
                }
                else
                {
                    _si.Main.LogError("خطا در ثبت سفارش");
                }
            }
            catch (Exception ex)
            {
                _si.Main.LogException("خطا در شبیه‌سازی عملیات دیتابیس", ex);
            }
            finally
            {
                _si.Main.LeaveMethod("DatabaseOperationSimulationExample");
            }
        }
